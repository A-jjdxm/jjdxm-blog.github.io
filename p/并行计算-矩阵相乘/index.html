static void matMultCPU_serial(const float* a, const float* b, float* c, int n)
{
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			double t = 0;
			for (int k = 0; k < n; k++)
			{
				t += (double)a[i * n + k] * b[k * n + j];
			}
			c[i * n + j] = t;
		}
	}
}

static void matMultCPU_parallel(const float* a, const float* b, float* c, int n)
{
#pragma omp parallel for schedule(dynamic)
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			double t = 0;
			for (int k = 0; k < n; k++)
			{
				t += (double)a[i * n + k] * b[k * n + j];
			}
			c[i * n + j] = t;
		}
	}
}


void genMat(float* arr, int n)
{
	int i, j;

	for (i = 0; i < n; i++)
	{
		for (j = 0; j < n; j++)
		{
			arr[i * n + j] = (float)rand() / RAND_MAX + (float)rand() / (RAND_MAX * RAND_MAX);
		}
	}
}

void matMultCPU(const float* a, const float* b, float* c, int n)
{
#pragma omp parallel for
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			double t = 0;
			for (int k = 0; k < n; k++)
			{
				t += (double)a[i * n + k] * b[k * n + j];
			}
			c[i * n + j] = t;
		}
	}
}

__global__ void matMultCUDAKernel1(const float* a, const float* b, float* c, int n)
{
	//è®¡ç®—è¿™ä¸ª thread åº”è¯¥è®¡ç®—çš„ row å’Œ col
	const int col = blockIdx.x * blockDim.x + threadIdx.x;
	const int row = blockIdx.y * blockDim.y + threadIdx.y;

	int i;
	//è®¡ç®—çŸ©é˜µä¹˜æ³• Kahanâ€™s Summation Formula
	if (row < n && col < n)
	{
		float t = 0;
		float y = 0;
		for (i = 0; i < n; i++)
		{
			float r;

			y -= a[row * n + i] * b[i * n + col];
			r = t - y;
			y = (r - t) + y;
			t = r;
		}
		c[row * n + col] = t;
	}
}

typedef struct Error {
	float max;
	float average;
}Error;

Error accuracyCheck(const float* a, const float* b, int n)
{
	Error err;
	err.max = 0;
	err.average = 0;
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			if (b[i * n + j] != 0)
			{
				//fabsæ±‚æµ®ç‚¹æ•°xçš„ç»å¯¹å€¼
				float delta = fabs((a[i * n + j] - b[i * n + j]) / b[i * n + j]);
				if (err.max < delta) err.max = delta;
				err.average += delta;
			}
		}
	}
	err.average = err.average / (n * n);
	return err;
}
